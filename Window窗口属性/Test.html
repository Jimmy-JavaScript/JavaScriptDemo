$(document).ready(function () { //offset 设置坐标值 $("button:first").click(function () { var txtTop = $("#txtTop").val(); var txtLeft = $("#txtLeft").val(); $(".box").offset({top: txtTop, left: txtLeft}); }); //offset 获取坐标值,距离页面上边、左边的距离，与父元素子元素的相对位置无关 $("button:eq(1)").click(function () { console.log("距离页面上部的距离：" + $(".box").offset().top); console.log("距离页面左部的距离：" + $(".box").offset().left); }); // position 获取坐标值，当父元素相对定位、绝对定位或固定定位时，利用position获得的时相对于父元素的上边和左边的距离 $("button:eq(2)").click(function () { console.log("距离页面上部的距离：" + $(".box").position().top); console.log("距离页面左部的距离：" + $(".box").position().left); }); });
